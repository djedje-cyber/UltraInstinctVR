151 static bool TryFindSample(string packageName, string packageVersion, string sampleDisplayName, out Sample sample)
152 {
153 sample = default;
154
155 if (!PackageVersionUtility.IsPackageInstalled(packageName))
156 return false;
157
158 IEnumerable<Sample> packageSamples;
159 try
160 {
161 packageSamples = Sample.FindByPackage(packageName, packageVersion);
162 }
163 catch (Exception e)
164 {
165 Debug.LogError($"Couldn't find samples of the {ToString(packageName, packageVersion)} package; aborting project validation rule. Exception: {e}");
166 return false;
167 }
168
169 if (packageSamples == null)
170 {
171 Debug.LogWarning($"Couldn't find samples of the {ToString(packageName, packageVersion)} package; aborting project validation rule.");
172 return false;
173 }
174
175 foreach (var packageSample in packageSamples)
176 {
177 if (packageSample.displayName == sampleDisplayName)
178 {
179 sample = packageSample;
180 return true;
181 }
182 }
183
184 Debug.LogWarning($"Couldn't find {sampleDisplayName} sample in the {ToString(packageName, packageVersion)} package; aborting project validation rule.");
185 return false;
186 }

193 static void InstallOrUpdateHands()
194 {
195 // Set a 3-second timeout for request to avoid editor lockup
196 var currentTime = DateTime.Now;
197 var endTime = currentTime + TimeSpan.FromSeconds(3);
198
199 var request = Client.Search(k_HandsPackageName);
200 if (request.Status == StatusCode.InProgress)
201 {
202 Debug.Log($"Searching for ({k_HandsPackageName}) in Unity Package Registry.");
203 while (request.Status == StatusCode.InProgress && currentTime < endTime)
204 currentTime = DateTime.Now;
205 }
206
207 var addRequest = k_HandsPackageName;
208 if (request.Status == StatusCode.Success && request.Result.Length > 0)
209 {
210 var versions = request.Result[0].versions;
211#if UNITY_2022_2_OR_NEWER
212 var recommendedVersion = new PackageVersion(versions.recommended);
213#else
214 var recommendedVersion = new PackageVersion(versions.verified);
215#endif
216 var latestCompatible = new PackageVersion(versions.latestCompatible);
217 if (recommendedVersion < s_RecommendedHandsPackageVersion && s_RecommendedHandsPackageVersion <= latestCompatible)
218 addRequest = $"{k_HandsPackageName}@{s_RecommendedHandsPackageVersion}";
219 }
220
221 s_HandsPackageAddRequest = Client.Add(addRequest);
222 if (s_HandsPackageAddRequest.Error != null)
223 {
224 Debug.LogError($"Package installation error: {s_HandsPackageAddRequest.Error}: {s_HandsPackageAddRequest.Error.message}");
225 }
226 }
