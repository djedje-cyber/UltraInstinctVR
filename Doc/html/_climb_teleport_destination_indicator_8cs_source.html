118 void OnClimbTeleportDestinationAnchorChanged(TeleportationMultiAnchorVolume teleportVolume)
119 {
120 HideIndicator();
121
122 var destinationAnchor = teleportVolume.destinationAnchor;
123 if (destinationAnchor == null)
124 return;
125
126 m_PointerInstance = Instantiate(m_PointerPrefab).transform;
127 var cameraTrans = teleportVolume.teleportationProvider.mediator.xrOrigin.Camera.transform;
128 var cameraPosition = cameraTrans.position;
129 var destinationPosition = destinationAnchor.position;
130 var destinationDirectionInScreenSpace = cameraTrans.InverseTransformDirection(destinationPosition - cameraPosition);
131 destinationDirectionInScreenSpace.z = 0f;
132 var pointerDirection = cameraTrans.TransformDirection(destinationDirectionInScreenSpace).normalized;
133 m_PointerInstance.position = destinationPosition - pointerDirection * m_PointerDistance;
134 m_PointerInstance.rotation = Quaternion.LookRotation(pointerDirection, -cameraTrans.forward);
135 }
