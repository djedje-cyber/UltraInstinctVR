219 void OnPokeStateDataUpdated(PokeStateData data)
220 {
221 var pokeTarget = data.target;
222 var applyFollow = m_ApplyIfChildIsTarget
223 ? pokeTarget != null && pokeTarget.IsChildOf(transform)
224 : pokeTarget == transform;
225
226 if (applyFollow)
227 {
228 var targetPosition = pokeTarget.InverseTransformPoint(data.axisAlignedPokeInteractionPoint);
229
230 if (m_ClampToMinDistance && targetPosition.sqrMagnitude < m_MinDistance * m_MinDistance)
231 targetPosition = Vector3.ClampMagnitude(targetPosition, m_MinDistance);
232
233 if (m_ClampToMaxDistance && targetPosition.sqrMagnitude > m_MaxDistance * m_MaxDistance)
234 targetPosition = Vector3.ClampMagnitude(targetPosition, m_MaxDistance);
235
236 m_TransformTweenableVariable.target = targetPosition;
237 m_PokeStrengthTweenableVariable.target = Mathf.Clamp01(data.interactionStrength);
238 }
239 else if (m_ReturnToInitialPosition)
240 {
241 m_TransformTweenableVariable.target = m_InitialPosition;
242 m_PokeStrengthTweenableVariable.target = 0f;
243 }
