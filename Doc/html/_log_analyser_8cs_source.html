18 void AnalyzeLogs()
19 {
20 if (!File.Exists(filePath))
21 {
22 Debug.LogError("Le fichier game_logs.txt n'existe pas !");
23 return;
24 }
25
26 int testCasesCount = 0;
27 int testFailedCount = 0;
28 int testPassedCount = 0;
29 Dictionary<string, OracleData> oracleData = new Dictionary<string, OracleData>();
30
31 foreach (string line in File.ReadLines(filePath))
32 {
33 if (line.Contains("TestGenerated")) testCasesCount++;
34 if (line.Contains("TestFailed")) testFailedCount++;
35 if (line.Contains("TestPassed")) testPassedCount++;
36
37 if (line.Contains("ORACLE"))
38 {
39 Match match = Regex.Match(line, @"ORACLE\s+(\S+)");
40 if (match.Success)
41 {
42 string oracleName = match.Groups[1].Value;
43 if (!oracleData.ContainsKey(oracleName))
44 oracleData[oracleName] = new OracleData();
45
46 oracleData[oracleName].Count++;
47 if (line.Contains("TestFailed"))
48 {
49 oracleData[oracleName].Failed++;
50 oracleData[oracleName].LogsFailed.Add(line);
51 }
52 if (line.Contains("TestPassed"))
53 {
54 oracleData[oracleName].Passed++;
55 oracleData[oracleName].LogsPassed.Add(line);
56 }
57 }
58 }
59 }
60
61 StringBuilder htmlReport = new StringBuilder();
62 htmlReport.AppendLine("<html><head><title>Unity Log Report</title>");
63 htmlReport.AppendLine("<style>");
64 htmlReport.AppendLine("body { font-family: Arial, sans-serif; margin: 20px; }");
65 htmlReport.AppendLine("h1 { color: #333; }");
66 htmlReport.AppendLine("table { border-collapse: collapse; width: 100%; }");
67 htmlReport.AppendLine("table, th, td { border: 1px solid black; }");
68 htmlReport.AppendLine("th, td { padding: 10px; text-align: center; }");
69 htmlReport.AppendLine("th { background-color: #f2f2f2; }");
70 htmlReport.AppendLine(".log-box { background: #f8f8f8; padding: 10px; border-radius: 5px; margin: 5px 0; }");
71 htmlReport.AppendLine(".log-passed { color: green; }");
72 htmlReport.AppendLine(".log-failed { color: red; }");
73 htmlReport.AppendLine("</style></head><body>");
74 htmlReport.AppendLine("<h1>Unity Log Analysis Report</h1>");
75 htmlReport.AppendLine("<h2>Summary</h2>");
76 htmlReport.AppendLine($"<p><strong>Nombre de logs contenant 'TestGenerated':</strong> {testCasesCount}</p>");
77 htmlReport.AppendLine($"<p><strong>Nombre de logs contenant 'TestFailed':</strong> {testFailedCount}</p>");
78 htmlReport.AppendLine($"<p><strong>Nombre de logs contenant 'TestPassed':</strong> {testPassedCount}</p>");
79
80 htmlReport.AppendLine("<h2>ORACLE Test Results</h2>");
81 htmlReport.AppendLine("<table>");
82 htmlReport.AppendLine("<tr><th>ORACLE</th><th>Instances</th><th>Test Passed</th><th>Test Failed</th></tr>");
83
84 foreach (var entry in oracleData)
85 {
86 htmlReport.AppendLine($"<tr><td>{entry.Key}</td><td>{entry.Value.Count}</td><td>{entry.Value.Passed}</td><td>{entry.Value.Failed}</td></tr>");
87 }
88 htmlReport.AppendLine("</table>");
89
90 foreach (var entry in oracleData)
91 {
92 htmlReport.AppendLine($"<h2>Logs pour ORACLE: {entry.Key}</h2>");
93 if (entry.Value.LogsPassed.Count > 0)
94 {
95 htmlReport.AppendLine("<h3 style='color: green;'>Tests Réussis</h3>");
96 foreach (string log in entry.Value.LogsPassed)
97 {
98 htmlReport.AppendLine($"<div class='log-box log-passed'>{log}</div>");
99 }
100 }
101 if (entry.Value.LogsFailed.Count > 0)
102 {
103 htmlReport.AppendLine("<h3 style='color: red;'>Tests Échoués</h3>");
104 foreach (string log in entry.Value.LogsFailed)
105 {
106 htmlReport.AppendLine($"<div class='log-box log-failed'>{log}</div>");
107 }
108 }
109 }
110 htmlReport.AppendLine("</body></html>");
111
112 File.WriteAllText("report_HTML.html", htmlReport.ToString(), Encoding.UTF8);
113 Debug.Log("The HTML report has been successfully generated: report_HTML.html");
114 }
115}
