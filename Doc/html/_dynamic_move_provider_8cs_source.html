115 protected override Vector3 ComputeDesiredMove(Vector2 input)
116 {
117 // Don't need to do anything if the total input is zero.
118 // This is the same check as the base method.
119 if (input == Vector2.zero)
120 return base.ComputeDesiredMove(input);
121
122 // Initialize the Head Transform if necessary, getting the Camera from XR Origin
123 if (m_HeadTransform == null)
124 {
125 var xrOrigin = mediator.xrOrigin;
126 if (xrOrigin != null)
127 {
128 var xrCamera = xrOrigin.Camera;
129 if (xrCamera != null)
130 m_HeadTransform = xrCamera.transform;
131 }
132 }
133
134 // Get the forward source for the left hand input
135 switch (m_LeftHandMovementDirection)
136 {
137 case MovementDirection.HeadRelative:
138 if (m_HeadTransform != null)
139 m_LeftMovementPose = m_HeadTransform.GetWorldPose();
140
141 break;
142
143 case MovementDirection.HandRelative:
144 if (m_LeftControllerTransform != null)
145 m_LeftMovementPose = m_LeftControllerTransform.GetWorldPose();
146
147 break;
148
149 default:
150 Assert.IsTrue(false, $"Unhandled {nameof(MovementDirection)}={m_LeftHandMovementDirection}");
151 break;
152 }
153
154 // Get the forward source for the right hand input
155 switch (m_RightHandMovementDirection)
156 {
157 case MovementDirection.HeadRelative:
158 if (m_HeadTransform != null)
159 m_RightMovementPose = m_HeadTransform.GetWorldPose();
160
161 break;
162
163 case MovementDirection.HandRelative:
164 if (m_RightControllerTransform != null)
165 m_RightMovementPose = m_RightControllerTransform.GetWorldPose();
166
167 break;
168
169 default:
170 Assert.IsTrue(false, $"Unhandled {nameof(MovementDirection)}={m_RightHandMovementDirection}");
171 break;
172 }
173
174 // Combine the two poses into the forward source based on the magnitude of input
175 var leftHandValue = leftHandMoveInput.ReadValue();
176 var rightHandValue = rightHandMoveInput.ReadValue();
177
178 var totalSqrMagnitude = leftHandValue.sqrMagnitude + rightHandValue.sqrMagnitude;
179 var leftHandBlend = 0.5f;
180 if (totalSqrMagnitude > Mathf.Epsilon)
181 leftHandBlend = leftHandValue.sqrMagnitude / totalSqrMagnitude;
182
183 var combinedPosition = Vector3.Lerp(m_RightMovementPose.position, m_LeftMovementPose.position, leftHandBlend);
184 var combinedRotation = Quaternion.Slerp(m_RightMovementPose.rotation, m_LeftMovementPose.rotation, leftHandBlend);
185 m_CombinedTransform.SetPositionAndRotation(combinedPosition, combinedRotation);
186
187 return base.ComputeDesiredMove(input);
188 }
189 }
