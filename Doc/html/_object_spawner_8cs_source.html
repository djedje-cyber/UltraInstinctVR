193 public bool TrySpawnObject(Vector3 spawnPoint, Vector3 spawnNormal)
194 {
195 if (m_OnlySpawnInView)
196 {
197 var inViewMin = m_ViewportPeriphery;
198 var inViewMax = 1f - m_ViewportPeriphery;
199 var pointInViewportSpace = cameraToFace.WorldToViewportPoint(spawnPoint);
200 if (pointInViewportSpace.z < 0f || pointInViewportSpace.x > inViewMax || pointInViewportSpace.x < inViewMin ||
201 pointInViewportSpace.y > inViewMax || pointInViewportSpace.y < inViewMin)
202 {
203 return false;
204 }
205 }
206
207 var objectIndex = isSpawnOptionRandomized ? Random.Range(0, m_ObjectPrefabs.Count) : m_SpawnOptionIndex;
208 var newObject = Instantiate(m_ObjectPrefabs[objectIndex]);
209 if (m_SpawnAsChildren)
210 newObject.transform.parent = transform;
211
212 newObject.transform.position = spawnPoint;
213 EnsureFacingCamera();
214
215 var facePosition = m_CameraToFace.transform.position;
216 var forward = facePosition - spawnPoint;
217 BurstMathUtility.ProjectOnPlane(forward, spawnNormal, out var projectedForward);
218 newObject.transform.rotation = Quaternion.LookRotation(projectedForward, spawnNormal);
219
220 if (m_ApplyRandomAngleAtSpawn)
221 {
222 var randomRotation = Random.Range(-m_SpawnAngleRange, m_SpawnAngleRange);
223 newObject.transform.Rotate(Vector3.up, randomRotation);
224 }
225
226 if (m_SpawnVisualizationPrefab != null)
227 {
228 var visualizationTrans = Instantiate(m_SpawnVisualizationPrefab).transform;
229 visualizationTrans.position = spawnPoint;
230 visualizationTrans.rotation = newObject.transform.rotation;
231 }
232
233 objectSpawned?.Invoke(newObject);
234 return true;
235 }
236 }
