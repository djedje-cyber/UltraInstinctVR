51 public static Bounds GetSceneBounds()
52 {
53 // Get all renderers in the scene
54 Renderer[] renderers = Object.FindObjectsByType<Renderer>(FindObjectsSortMode.None);
55
56 if (renderers.Length == 0)
57 {
58 Debug.LogWarning("No renderers found in the scene.");
59 return new Bounds(Vector3.zero, Vector3.zero);
60 }
61
62 // Initialize bounds to the first renderer
63 Bounds sceneBounds = renderers[0].bounds;
64
65 // Encapsulate all other renderers
66 for (int i = 1; i < renderers.Length; i++)
67 {
68 sceneBounds.Encapsulate(renderers[i].bounds);
69 }
70
71 return sceneBounds;
72 }
