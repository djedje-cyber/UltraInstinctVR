85 // Smooths the hand movements of an XRHandSubsystem by updating the root
86 // pose of the left and right hands with interpolated positions.
87 void SmoothHandsExample(XRHandSubsystem subsystem, XRHandSubsystem.UpdateSuccessFlags successFlags, XRHandSubsystem.UpdateType updateType, bool modeChanged)
88 {
89 var leftHand = subsystem.leftHand;
90 var rightHand = subsystem.rightHand;
91
92 if (leftHand.isTracked && m_LeftHandSmoothingFactor > 0)
93 {
94 var leftPose = leftHand.rootPose;
95 var currentLeftHandPosition = leftPose.position;
96 if (!m_FirstFrame && !modeChanged)
97 {
98 float tweenAmt = Time.deltaTime * m_LeftHandSmoothingFactor;
99 currentLeftHandPosition = Vector3.Lerp(m_LastLeftHandPosition, currentLeftHandPosition, tweenAmt);
100 m_LeftHandPose.position = currentLeftHandPosition;
101 m_LeftHandPose.rotation = leftPose.rotation;
102
103 leftHand.SetRootPose(m_LeftHandPose);
104 subsystem.SetCorrespondingHand(leftHand);
105 }
106 m_LastLeftHandPosition = currentLeftHandPosition;
107 }
108
109 if (rightHand.isTracked && m_RightHandSmoothingFactor > 0)
110 {
111 var rightPose = rightHand.rootPose;
112 var currentRightHandPosition = rightPose.position;
113 if (!m_FirstFrame && !modeChanged)
114 {
115 float tweenAmt = Time.deltaTime * m_RightHandSmoothingFactor;
116 currentRightHandPosition = Vector3.Lerp(m_LastRightHandPosition, currentRightHandPosition, tweenAmt);
117 m_RightHandPose.position = currentRightHandPosition;
118 m_RightHandPose.rotation = rightPose.rotation;
119
120 rightHand.SetRootPose(m_RightHandPose);
121 subsystem.SetCorrespondingHand(rightHand);
122 }
123 m_LastRightHandPosition = currentRightHandPosition;
124 }
125 }
