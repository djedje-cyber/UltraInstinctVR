200 void SetGestureState(SystemGestureState state, bool forceInvoke)
201 {
202 if (!forceInvoke && m_SystemGestureState.Value == state)
203 return;
204
205 m_SystemGestureState.Value = state;
206 switch (state)
207 {
208 case SystemGestureState.Ended:
209 m_SystemGestureEnded?.Invoke();
210 break;
211 case SystemGestureState.Started:
212 m_SystemGestureStarted?.Invoke();
213 break;
214 }

217#if XR_HANDS_1_1_OR_NEWER && (ENABLE_VR || UNITY_GAMECORE)
218 void UpdateAimFlags(MetaAimFlags value, bool forceInvoke = false)
219 {
220 var hadMenuPressed = (m_AimFlags & MetaAimFlags.MenuPressed) != 0;
221 m_AimFlags = value;
222 var hasSystemGesture = (m_AimFlags & MetaAimFlags.SystemGesture) != 0;
223 var hasMenuPressed = (m_AimFlags & MetaAimFlags.MenuPressed) != 0;
224 var hasValid = (m_AimFlags & MetaAimFlags.Valid) != 0;
225 var hasIndexPinching = (m_AimFlags & MetaAimFlags.IndexPinching) != 0;
226
227 if (!hadMenuPressed && hasMenuPressed)
228 {
229 m_MenuPressed?.Invoke();
230 }
231
232 if (hasSystemGesture || hasMenuPressed)
233 {
234 SetGestureState(SystemGestureState.Started, forceInvoke);
235 return;
236 }
237
238 if (hasValid)
239 {
240 SetGestureState(SystemGestureState.Ended, forceInvoke);
241 return;
242 }
243
244 // We want to keep the system gesture going when the user is still index pinching
245 // even though the SystemGesture flag is no longer set.
246 if (hasIndexPinching && m_SystemGestureState.Value != SystemGestureState.Ended)
247 {
248 SetGestureState(SystemGestureState.Started, forceInvoke);
249 return;
250 }
251
252 SetGestureState(SystemGestureState.Ended, forceInvoke);
253 }
254#endif
