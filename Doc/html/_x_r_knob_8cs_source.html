260 void UpdateRotation(bool freshCheck = false)
261 {
262 // Are we in position offset or direction rotation mode?
263 var interactorTransform = m_Interactor.GetAttachTransform(this);
264
265 // We cache the three potential sources of rotation - the position offset, the forward vector of the controller, and up vector of the controller
266 // We store any data used for determining which rotation to use, then flatten the vectors to the local xz plane
267 var localOffset = transform.InverseTransformVector(interactorTransform.position - m_Handle.position);
268 localOffset.y = 0.0f;
269 var radiusOffset = transform.TransformVector(localOffset).magnitude;
270 localOffset.Normalize();
271
272 var localForward = transform.InverseTransformDirection(interactorTransform.forward);
273 var localY = Math.Abs(localForward.y);
274 localForward.y = 0.0f;
275 localForward.Normalize();
276
277 var localUp = transform.InverseTransformDirection(interactorTransform.up);
278 localUp.y = 0.0f;
279 localUp.Normalize();
280
281 if (m_PositionDriven && !freshCheck)
282 radiusOffset *= (1.0f + k_ModeSwitchDeadZone);
283
284 // Determine when a certain source of rotation won't contribute - in that case we bake in the offset it has applied
285 // and set a new anchor when they can contribute again
286 if (radiusOffset >= m_PositionTrackedRadius)
287 {
288 if (!m_PositionDriven || freshCheck)
289 {
290 m_PositionAngles.SetBaseFromVector(localOffset);
291 m_PositionDriven = true;
292 }
293 }
294 else
295 m_PositionDriven = false;
296
297 // If it's not a fresh check, then we weight the local Y up or down to keep it from flickering back and forth at boundaries
298 if (!freshCheck)
299 {
300 if (!m_UpVectorDriven)
301 localY *= (1.0f - (k_ModeSwitchDeadZone * 0.5f));
302 else
303 localY *= (1.0f + (k_ModeSwitchDeadZone * 0.5f));
304 }
305
306 if (localY > 0.707f)
307 {
308 if (!m_UpVectorDriven || freshCheck)
309 {
310 m_UpVectorAngles.SetBaseFromVector(localUp);
311 m_UpVectorDriven = true;
312 }
313 }
314 else
315 {
316 if (m_UpVectorDriven || freshCheck)
317 {
318 m_ForwardVectorAngles.SetBaseFromVector(localForward);
319 m_UpVectorDriven = false;
320 }
321 }
322
323 // Get angle from position
324 if (m_PositionDriven)
325 m_PositionAngles.SetTargetFromVector(localOffset);
326
327 if (m_UpVectorDriven)
328 m_UpVectorAngles.SetTargetFromVector(localUp);
329 else
330 m_ForwardVectorAngles.SetTargetFromVector(localForward);
331
332 // Apply offset to base knob rotation to get new knob rotation
333 var knobRotation = m_BaseKnobRotation - ((m_UpVectorAngles.totalOffset + m_ForwardVectorAngles.totalOffset) * m_TwistSensitivity) - m_PositionAngles.totalOffset;
334
335 // Clamp to range
336 if (m_ClampedMotion)
337 knobRotation = Mathf.Clamp(knobRotation, m_MinAngle, m_MaxAngle);
338
339 SetKnobRotation(knobRotation);
340
341 // Reverse to get value
342 var knobValue = (knobRotation - m_MinAngle) / (m_MaxAngle - m_MinAngle);
343 SetValue(knobValue);
344 }

347 {

358 void SetValue(float newValue)
359 {
360 if (m_ClampedMotion)
361 newValue = Mathf.Clamp01(newValue);
362
363 if (m_AngleIncrement > 0)
364 {
365 var angleRange = m_MaxAngle - m_MinAngle;
366 var angle = Mathf.Lerp(0.0f, angleRange, newValue);
367 angle = Mathf.Round(angle / m_AngleIncrement) * m_AngleIncrement;
368 newValue = Mathf.InverseLerp(0.0f, angleRange, angle);
369 }
370
371 m_Value = newValue;
372 m_OnValueChange.Invoke(m_Value);
373 }

385 static float ShortestAngleDistance(float start, float end, float max)
386 {
387 var angleDelta = end - start;
388 var angleSign = Mathf.Sign(angleDelta);
389
390 angleDelta = Math.Abs(angleDelta) % max;
391 if (angleDelta > (max * 0.5f))
392 angleDelta = -(max - angleDelta);
393
394 return angleDelta * angleSign;
395 }

397 void OnDrawGizmosSelected()
398 {
399 const int k_CircleSegments = 16;
400 const float k_SegmentRatio = 1.0f / k_CircleSegments;
401
402 // Nothing to do if position radius is too small
403 if (m_PositionTrackedRadius <= Mathf.Epsilon)
404 return;
405
406 var knobTransform = transform;
407
408 // Draw a circle from the handle point at size of position tracked radius
409 var circleCenter = knobTransform.position;
410
411 if (m_Handle != null)
412 circleCenter = m_Handle.position;
413
414 var circleX = knobTransform.right;
415 var circleY = knobTransform.forward;
416
417 Gizmos.color = Color.green;
418 var segmentCounter = 0;
419 while (segmentCounter < k_CircleSegments)
420 {
421 var startAngle = segmentCounter * k_SegmentRatio * 2.0f * Mathf.PI;
422 segmentCounter++;
423 var endAngle = segmentCounter * k_SegmentRatio * 2.0f * Mathf.PI;
424
425 Gizmos.DrawLine(circleCenter + (Mathf.Cos(startAngle) * circleX + Mathf.Sin(startAngle) * circleY) * m_PositionTrackedRadius,
426 circleCenter + (Mathf.Cos(endAngle) * circleX + Mathf.Sin(endAngle) * circleY) * m_PositionTrackedRadius);
427 }
